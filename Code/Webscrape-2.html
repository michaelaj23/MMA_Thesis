                    gate_cell = row.find('td')

                    buyrate_cell = row.find('td')
                    buyrate = buyrate_cell.text.strip() if buyrate_cell else 'Not available'
            data = data.append({'Title': title, 'Date': date, 'Total Gate': gate, 'Buyrate': buyrate}, ignore_index=True)
